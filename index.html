import pandas as pd
import plotnine
from plotnine import ggplot, aes

DF = pd.read_csv("Optimization Data.csv")
DF["Flexural Stiffness (N/mm)"] = 0
DF["Torsional Stiffness (Nm/mm)"] = 0
DF["Compressive Stiffness (N/mm)"] = 0
DF["Flexural Stiffness (N/mm)"] = DF["Force"]/DF["Max Deformation (mm)"]
DF["Torsional Stiffness (Nm/mm)"] = DF["Moment"]/DF["Max Deformation (mm)"]
DF["Compressive Stiffness (N/mm)"] = DF["Force"]/DF["Max Deformation (mm)"]
DF["Enclosed Density (kg/m3)"] = DF["Weight (kg)"]/(12570000*10**(-9))
DF["Flexural Stiffness-to-Weight (N/mm-kg)"] = DF["Flexural Stiffness (N/mm)"]/DF["Weight (kg)"]
DF["Torsional Stiffness-to-Weight (Nm/mm-kg)"] = DF["Torsional Stiffness (Nm/mm)"]/DF["Weight (kg)"]


print(ggplot() + 
  plotnine.geom_point(data=DF[DF["Loading"]=="Flexure"], 
  mapping=aes(x="Enclosed Density (kg/m3)", 
  y="Max Deformation (mm)", 
  color="Material", 
  shape="Geometry")) + 
  plotnine.labs(title='Design Flexure Deformation and Weight'))
print(ggplot() + 
  plotnine.geom_point(data=DF[DF["Loading"]=="Torsion"], 
  mapping=aes(x="Enclosed Density (kg/m3)", 
  y="Max Deformation (mm)", 
  color="Material", 
  shape="Geometry")) + 
  plotnine.labs(title='Design Torsion Deformation and Weight'))
print(ggplot() + 
  plotnine.geom_point(data=DF[DF["Loading"]=="Compression"], 
  mapping=aes(x="Enclosed Density (kg/m3)", 
  y="Max Deformation (mm)", 
  color="Material", 
  shape="Geometry")) + 
  plotnine.labs(title='Design Compression Deformation and Weight'))
